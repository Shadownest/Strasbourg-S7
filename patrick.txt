CONFIGURATION
git config --global color.diff auto
git config --global color.status auto
git config --global push.default current
git config --global pull.default current
git config --global color.branch auto
git config --global user.email "YOUREMAIL"
git config --global user.name "YOURLOGIN"

INITIALISATION
cd /le/chemin/de/votre/site
git init
git remote add origin VOTRE_URL
git remote -v
git fetch
git checkout master
git pull

UTILISATION
git add monfichier.txt
git commit -m "Mon message" monfichier.txt
git pull
git push
 
// Lors de l'ajout d'un fichier :
git add FILE
// Si jamais vous savez pas où vous en êtes :
git status

BRANCHING
// Créer une branche
git branch NOM_DE_LA_BRANCHE
// Changer de branche
git checkout NOM_DE_LA_BRANCHE
// Combiner la branche actuelle avec une autre
git merge NOM_DE_LA_BRANCHE
// Supprimer une branche après utilisation
git branch -d NOM_DE_LA_BRANCHE
